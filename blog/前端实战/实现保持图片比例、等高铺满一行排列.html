# 实现保持图片比例、等高铺满一行排列

其实这也是面试的时候所遇到的一道题，当时是只给出了大概的思路。现在有时间整理一下思路顺便写一下实现。其实所想要的这个效果就是百度搜索图片的时候的一个效果：

场景就是一个搜索图片的场景，图片已经知道宽高，最终渲染出的图片的顺序不能改变。图片的纵横比也不能改变（就是不能被压变形）还有图片不允许被裁剪，然后效果就是呈现出来的这种效果。

首先可以确定的一点是，一行渲染出来的图片宽度就等于容器的宽度。但是问题在于如何去决定一行渲染几张图片？ 如果一行是渲染四张图片，那么可以很轻松的计算出每张图片此时的宽度。但是要保持图片的纵横比又不能裁剪图片的话，高度不能保证是一致的，除非每张图片的纵横比都相等。但是这种美妙的场景很显然不会出现。

面试官看我面露难色，问我需不需要给出一点提示，我说好吧。然后面试官说如果固定一个初始高度的话。。。这一句话如醍醐灌顶，电光火石之间，一个想法就在心里出来了。

首先，容器的宽度是很容易知道的。比如如果初设的高度是300，图片原始的宽高也是知道的，那么就可以计算出300与之对应的高度。按照这样的想法先开始排第一排，比如现在我们想每一行的总的宽度是1000。每一张图片的宽高都不一样（但是是已知的），现在我们限定了图片高度，那么根据纵横比就能算出图片的宽度。按照这样排下去，总会排到一张图片这时候的宽度加上之前已经排好的宽度大于了1000。这个时候，就要决定要不要留下。

如图所示，排到第四张图片的时候这时候四张图片总的宽度已经超过容器宽度了，如何决定第四张图片要不要留在第一行呢？首先，要确保一行的图片宽度加起来能够刚好等于容器的宽度，这个时候就需要去调整图片的高度以此改变图片的宽度。在看实际场景，如果要留下第四张图片，那么此时需要缩小图片高度，如果要移除第四张图片，那么就要增加图片高度。在看第四张图片超过的部分是小于自身的一半的，这意味着保留图片比一处图片所需要调整的宽度要小。所以选择保留第四张图片。

思路有了写下来是代码实现：

<!DOCTYPE html>
<html>
<head>
	<title>图片等高排列</title>
	<style type="text/css">
		* {
			padding: 0;
			margin: 0;
		}
		.item {
			display: inline-block;
			background-color: blue;
			box-sizing: border-box;
			border: 1px solid #fff;
			vertical-align: bottom;
			margin-top: 6px;
			margin-left: 6px;
		}
	</style>
</head>
<body>
	<div id="container">
		
	</div>
	<script type="text/javascript" src="./imgList.js"></script>
	<script type="text/javascript">
		

		let containerDom = document.getElementById('container')

		const initHeight = 300
		const sumWidth = containerDom.offsetWidth - 3 // 减 3 留一点误差的空间，以免元素换行
		const margin = 6 // 图片之间的间隙

		let arr = getImgList(imgList)
		let divDom = document.createElement('div')

		arr.forEach(item=>{
			let imgDom = document.createElement('div')
			imgDom.style.width = item.width + 'px'
			imgDom.style.height = item.height + 'px'
			imgDom.className = 'item'
			divDom.appendChild(imgDom)
		})

		containerDom.appendChild(divDom)

		function getImgList(list){
			let arr = []
			let index = 0
			while(index < list.length){
				let row = computedRow(index, list)
				console.log('row', row)
				arr = arr.concat(row)
				index += row.length
			}
			return arr
		}

		function computedRow(startIndex, list){
			let ret = []
			let index = startIndex
			let recordWidth = 0

			let offset

			while( index < list.length ){
				let item = list[index]
				let currentWidth = Math.floor((item.width * initHeight) / item.height)
				recordWidth += currentWidth
				ret.push({
					thumbURL: item.thumbURL,
					width: currentWidth,
					height: initHeight
				})
				if ( recordWidth >= sumWidth ) {
					if (  recordWidth - sumWidth > (currentWidth/2) ) {
						index--
						ret.pop()
						recordWidth -= currentWidth
					}
					break
				}
				index++
			}
			
			// 这里的ret就是将要形成一行的数据

			offset = recordWidth - sumWidth

			// 这里传递给 recomputedWidth 时的总的宽度将 元素间隙减出来。这样才能在有间隙的情况下，
			// 算出正确的高度
			ret = recomputedWidth(ret, offset, initHeight, sumWidth-(ret.length * margin))
			return ret
		}


		function recomputedWidth(list, offset, prevHeight, sumWidth){
			let newHeight = offset > 0 ? prevHeight-1 : prevHeight + 1
			let recordWidth = 0
			let ret = []
			ret = list.map(item=>{
				let currentWidth = Math.floor((item.width * newHeight) / item.height)
				recordWidth += currentWidth
				return {
					...item,
					width: currentWidth,
					height: newHeight
				}
			})
			offset = recordWidth - sumWidth
			if ( Math.abs(offset) > 2 ) return recomputedWidth(ret, recordWidth - sumWidth, newHeight, sumWidth)
			return ret		
		}

	</script>
</body>
</html>
